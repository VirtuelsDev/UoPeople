% Définition du style de page personnalisé pour la première page
\fancypagestyle{firstpage}{
    \fancyhf{} % Efface les en-têtes et pieds de page par défaut
    \renewcommand{\headrulewidth}{0.4pt} % Épaisseur de la ligne d'en-tête
    \renewcommand{\footrulewidth}{0.4pt} % Épaisseur de la ligne de pied de page
    \fancyhead[L]{\href{\urlleft}{En-tête gauche\\ https://www.example.com}} % Texte à gauche de l'en-tête
    \fancyhead[R]{En-tête droite} % Texte à droite de l'en-tête
    %\fancyhead[R]{\includegraphics[height=20pt]{\logo}}
    \fancyfoot[L]{Pied de page gauche} % Texte à gauche du pied de page
    \fancyfoot[R]{\thepage~|~\authorname} % Page number | Author of the footer
}

% Définition du style de page pour les pages paires (même pages)
\fancypagestyle{evenpages}{
    \fancyhf{} % Efface les en-têtes et pieds de page par défaut
    \renewcommand{\headrulewidth}{0pt} % Supprime la ligne d'en-tête de page
    \renewcommand{\footrulewidth}{0.4pt} % Épaisseur de la ligne de pied de page
    \fancyfoot[L]{Pied de page gauche} % Texte à gauche du pied de page
    \fancyfoot[R]{\thepage~|~\authorname} % Page number | Author of the footer
}

% Définition du style de page pour les pages impaires (pages impaires)
\fancypagestyle{oddpages}{
   \fancyhf{} % Efface les en-têtes et pieds de page par défaut
    \renewcommand{\headrulewidth}{0pt} % Supprime la ligne d'en-tête de page
    \renewcommand{\footrulewidth}{0.4pt} % Épaisseur de la ligne de pied de page
    \fancyfoot[L]{\volum\ \issue\ | \thepage} % Footer content for odd pages
    \fancyfoot[R]{\thepage~|~\authorname} % Page number | Author of the footer
}
